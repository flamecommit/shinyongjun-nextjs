React Fullpage

This is a Fullpage library that operates in React.

Issues
NPM

Installation and usage

npm install @shinyongjun/react-fullpage
yarn add @shinyongjun/react-fullpage

There are two types of components: FullpageContainer and FullpageSection.

import { useState } from 'react';
import {
  FullpageContainer,
  FullpageSection,
} from '@shinyongjun/react-fullpage';

function ExampleComponent() {
  const [activeIndex, setActiveIndex] = useState<number>(0);

  return (
    <FullpageContainer
      activeIndex={activeIndex}
      setActiveIndex={setActiveIndex}
    >
      <FullpageSection>
        <div>Section 1</div>
      </FullpageSection>
      <FullpageSection>
        <div>Section 2</div>
      </FullpageSection>
      <FullpageSection>
        <div>Section 3</div>
      </FullpageSection>
      <FullpageSection isAutoHeight>
        <footer>Footer</footer>
      </FullpageSection>
    </FullpageContainer>
  );
}

export default ExampleComponent;

FullpageContainer Props

activeIndex

required : true
Type : number
Description : index of activeSection.

setActiveIndex

required : true
Type : function
Description : set activeIndex.

import { useState } from 'react';
import {
  FullpageContainer,
  FullpageSection,
} from '@shinyongjun/react-fullpage';

function ExampleComponent() {
  const [activeIndex, setActiveIndex] = useState<number>(0);

  return (
    <>
      <FullpageContainer
        activeIndex={activeIndex}
        setActiveIndex={setActiveIndex}
      >
        <FullpageSection>
          <div>Section 1</div>
        </FullpageSection>
        <FullpageSection>
          <div>Section 2</div>
        </FullpageSection>
        <FullpageSection>
          <div>Section 3</div>
        </FullpageSection>
        <FullpageSection isAutoHeight>
          <footer>Footer</footer>
        </FullpageSection>
      </FullpageContainer>
      <div className="controller">
        <button
          type="button"
          className={`${activeIndex === 0 ? 'active' : ''}`}
          onClick={() => setActiveIndex(0)}
        >
          1
        </button>
        <button
          type="button"
          className={`${activeIndex === 1 ? 'active' : ''}`}
          onClick={() => setActiveIndex(1)}
        >
          2
        </button>
        <button
          type="button"
          className={`${activeIndex === 2 ? 'active' : ''}`}
          onClick={() => setActiveIndex(2)}
        >
          3
        </button>
      </div>
    </>
  );
}

export default ExampleComponent;

onBeforeChange

Argument : beforeIndex: number, afterIndex: number
Description : The callback that runs when you before a transition.

onAfterChange

Argument : beforeIndex: number, afterIndex: number
Description : The callback that runs when you after a transition.

import { useState } from 'react';
import {
  FullpageContainer,
  FullpageSection,
} from '@shinyongjun/react-fullpage';

function ExampleComponent() {
  const [activeIndex, setActiveIndex] = useState<number>(0);

  return (
    <FullpageContainer
      activeIndex={activeIndex}
      setActiveIndex={setActiveIndex}
      onBeforeChange={(beforeIndex, afterIndex) => {
        console.log('before', beforeIndex, afterIndex);
      }}
      onAfterChange={(beforeIndex, afterIndex) => {
        console.log('after', beforeIndex, afterIndex);
      }}
    >
      <FullpageSection>
        <div>Section 1</div>
      </FullpageSection>
      <FullpageSection>
        <div>Section 2</div>
      </FullpageSection>
      <FullpageSection>
        <div>Section 3</div>
      </FullpageSection>
      <FullpageSection isAutoHeight>
        <footer>Footer</footer>
      </FullpageSection>
    </FullpageContainer>
  );
}

export default ExampleComponent;

FullpageSection Props

name

Type : string
Description : When rendering, it goes to the name section that matches the hash(#).
(After rendering, hash disappears from url.)

isAutoHeight

Type : boolean
Description : When true, the height value is auto. (Used primarily for Footer.)
