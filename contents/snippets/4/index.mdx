```ts
const addCommas = (value: string | number): string => {
  const formatIntegerPart = (integerPart: string): string =>
    integerPart.replace(/\B(?=(\d{3})+(?!\d))/g, ',');

  const [integerPart, decimalPart = ''] = value.toString().split('.');
  return `${formatIntegerPart(integerPart)}${
    decimalPart ? `.${decimalPart}` : ''
  }`;
};

console.log(addCommas(651651)); // 651,651
console.log(addCommas(651651.3213)); // 651,651.3213
console.log(addCommas('1534.3213')); // 1,534.3213
console.log(addCommas(-154651651.3213)); // -154,651,651.3213
```
