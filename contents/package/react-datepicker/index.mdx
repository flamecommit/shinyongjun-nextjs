## Installation

```
npm install @shinyongjun/react-datepicker
yarn add @shinyongjun/react-datepicker
```

## Usage

```tsx
import { Datepicker } from '@shinyongjun/react-datepicker';

function ExampleComponent() {
  return <Datepicker />;
}

export default ExampleComponent;
```
