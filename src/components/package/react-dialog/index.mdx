### Alert Example

#### - code

```tsx
import { useState } from 'react';
import { useDialog } from '@shinyongjun/react-dialog';
import '@shinyongjun/react-dialog/css';

function AlertExample() {
  const { alert } = useDialog();
  const [state, setState] = useState('close');

  const handleAlert = async () => {
    setState('open');
    await alert('Alert message.');
    setState('close');
  };

  return (
    <>
      <button type="button" onClick={handleAlert}>
        Alert
      </button>
      <div>state : {state}</div>
    </>
  );
}
```

### Confirm Example

#### - code

```tsx
import { useState } from 'react';
import { useDialog } from '@shinyongjun/react-dialog';
import '@shinyongjun/react-dialog/css';

function ConfirmExample() {
  const { confirm } = useDialog();
  const [result, setResult] = useState<string>('');

  const handleConfirm = async () => {
    setResult('');
    const r = await confirm('Confirm message.');
    setResult(r.toString());
  };

  return (
    <>
      <button type="button" onClick={handleConfirm}>
        Confirm
      </button>
      <span>result : {result}</span>
    </>
  );
}
```

### Prompt Example

#### - code

```tsx
import { useState } from 'react';
import { useDialog } from '@shinyongjun/react-dialog';
import '@shinyongjun/react-dialog/css';

function PromptExample() {
  const { prompt } = useDialog();
  const [result, setResult] = useState<string | null>(null);

  const handlePrompt = async () => {
    setResult(null);
    const r = await prompt('Prompt message.');
    setResult(r);
  };

  return (
    <>
      <button type="button" onClick={handlePrompt}>
        Prompt
      </button>
      <span>result : {result}</span>
    </>
  );
}
```
